#!/usr/bin/env python3
import sys

for line in sys.stdin:
    line = line.strip()
    if line:
        try:
            parts = line.split(',')
            
            # Skip header
            if parts[0] == "order_id":
                continue
            
            item_name = parts[2]
            quantity = int(parts[1])
            
            print(f"{item_name}\t{quantity}")
        
        except:
            continue
